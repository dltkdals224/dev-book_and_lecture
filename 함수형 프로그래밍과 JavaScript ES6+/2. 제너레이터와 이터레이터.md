# 2. 제너레이터와 이터레이터

## 제너레이터와 이터레이터

제너레이터는 이터레이터이자 이터러블을 생성하는 함수.<br/>
즉, 이터레이터를 리턴하는 함수.

```javascript
function* gener() {
  yield 1;
  yield 2;
  yield 3;
  return 100;
  // return은 yield와 달리 done: true가 된다.
  // return은 순회문에서는 탐색되지 않는다.
}
let iter = gener();

console.log(iter[Symbol.iterator]() == iter); // true

console.log(iter.next()); // {value: 1, done: false}
console.log(iter.next()); // {value: 2, done: false}
console.log(iter.next()); // {value: 3, done: false}
console.log(iter.next()); // {value: 100, done: true}
console.log(iter.next()); // {value: undefined, done: true}

for (const a of gener()) {
  console.log(a); // 1 2 3
}

// 제너레이터는 순회 대상 리스트를 문(장)으로 만드는 것이라고 생각하면 된다.
// (중요)제너레이터를 통해 어떠한 상태, 값 이라도 순회할 수 있도록 만들 수 있다.
```

## odds

제너레이터의 중첩 사용 예시.

```javascript
function* infinity(i = 0) {
  while (true) {
    yield i++;
  }
}
function* limit(limit, iter) {
  for (const a of iter) {
    yield a;
    if (a === limit) {
      return;
    }
  }
}

function* odds(MAX_VALUE) {
  for (const a of limit(MAX_VALUE, infinity(1))) {
    if (a % 2) {
      yield a;
    }
  }
}
let iter = odds(10);
console.log(iter.next()); // {value: 1, done: false}
console.log(iter.next()); // {value: 3, done: false}
console.log(iter.next()); // {value: 5, done: false}
console.log(iter.next()); // {value: 7, done: false}
console.log(iter.next()); // {value: 9, done: false}
console.log(iter.next()); // {value: undefined, done: true}

for (const a of odds(40)) {
  console.log(a); // 1 3 5 ... 35 37 39
}
```

## for...of, 전재 연산자, 구조 분해, 나머지 연산자

```javascript
// function* infinity(i = 0)
// function* limit(limit, iter)
// function* odds(MAX_VALUE)

// 세 함수 위와 동일

console.log([...ods(10), ...odds(20)]); // [1, 3, 5, ... ,15, 17, 19]

const [head, ...tail] = odds(5);
console.log(head); // 1
console.log(tail); // [3, 5]

const [a, b, ...rest] = odds(10);
console.log(a); // 1
console.log(b); // 3
console.log(rest); // [5, 7, 9]
```
